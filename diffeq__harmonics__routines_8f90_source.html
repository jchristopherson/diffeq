113module function frf_sweep(sys, freq, iv, solver, ncycles, ntransient, &
114 pointspercycle, err) result(rst)
115 ! Arguments
116 class(harmonic_ode_container), intent(inout) :: sys
117 real(real64), intent(in), dimension(:) :: freq, iv
118 class(ode_integrator), intent(inout), optional, target :: solver
119 integer(int32), intent(in), optional :: ncycles, ntransient, pointspercycle
120 class(errors), intent(inout), optional, target :: err
121 complex(real64), allocatable, dimension(:,:) :: rst

124 integer(int32) :: i, j, nfreq, neqn, nc, nt, ntotal, npts, ppc, flag, lwork
125 real(real64) :: dt
126 real(real64), allocatable, dimension(:) :: ic, t, fftwork

134 ! Initialization
135 if (present(solver)) then
136 integrator => solver
137 else
138 integrator => default_integrator
139 end if
140 if (present(ncycles)) then
141 nc = ncycles
142 else
143 nc = 5
144 end if
145 if (present(ntransient)) then
146 nt = ntransient
147 else
148 nt = 30
149 end if
150 if (present(pointspercycle)) then
151 ppc = pointspercycle
152 else
153 ppc = 30
154 end if
155 if (present(err)) then
156 errmgr => err
157 else
158 errmgr => deferr
159 end if
160 nfreq = size(freq)
161 neqn = size(iv)
162 ntotal = nt + nc
163 npts = ntotal * ppc
164 lwork = 2 * nc * ppc + 15
165
166 ! Input Checking
167
168 ! Local Memory Allocation
169 allocate(rst(nfreq, neqn), stat = flag)
170 if (flag == 0) allocate(ic(neqn), stat = flag, source = iv)
171 if (flag == 0) allocate(t(npts), stat = flag)
172 if (flag == 0) allocate(fftwork(lwork), stat = flag)
173 if (flag /= 0) go to 10
174
175 ! Set up the Fourier transform
176 call dffti(nc * ppc, fftwork)
177
178 ! Cycle over each frequency point
179 do i = 1, nfreq
180 ! Define the time vector
181 dt = ntotal / freq(i)
182 t = (/ (dt * j, j = 0, npts - 1) /)
183
184 ! Update the frequency
185 sys%excitation_frequency = freq(i)
186
187 ! Compute the solution
188 sol = integrator%solve(sys, t, ic, errmgr)
189 if (errmgr%has_error_occurred()) return
190
191 ! Reset the initial conditions to the last solution point
192 ic = sol(npts, 2:)
193
194 ! Determine the magnitude and phase for each equation
195 do j = 1, neqn
196 rst(i,j) = get_magnitude_phase(sol(nt*ppc+1:,j+1), fftwork)
197 end do
198 end do
199
200 ! End
201 return

20410 continue

209function get_magnitude_phase(x, work) result(rst)
210 ! Arguments
211 real(real64), intent(inout), dimension(:) :: x, work
212 complex(real64) :: rst
213
214 ! Local Variables
215 integer(int32) :: i, n, m
216 real(real64) :: mag, check, scale, w, sumw
217 complex(real64) :: val
218 type(hamming_window) :: win
219
220 ! Initialization
221 n = size(x)
222 win%size = n
223 m = compute_transform_length(n)
224 if (mod(n, 2) == 0) then
225 scale = 2.0d0 / n
226 else
227 scale = 2.0d0 / (n - 1.0d0)
228 end if
229
230 ! Window the data
231 sumw = 0.0d0
232 do i = 1, n
233 w = win%evaluate(i - 1)
234 x(i) = w * x(i)
235 sumw = sumw + w
236 end do
237 scale = scale * n / sumw
238
239 ! Transform the data and find the largest magnitude component
240 call dfftf(n, x, work)
241 check = 0.0d0
242 do i = 2, m
243 val = scale * cmplx(x(2*i-1), x(2*i), real64)
244 mag = abs(val)
245 if (mag > check) then
246 rst = val
247 check = mag
248 end if
249 end do
250end function
